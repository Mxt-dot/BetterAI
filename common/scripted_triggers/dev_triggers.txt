power_tech = {
	$power$_power = $value$
		variable_arithmetic_trigger = {
			export_to_variable = {
				variable_name = power
				value = trigger_value:$power$_power
				who = PREV
			}
			export_to_variable = {
				variable_name = powerCost
				value = modifier:all_power_cost
				who = PREV
			}
			change_variable = {
				which = powerCost
				value = 1
			}
			OR = { PREV = { has_institution = feudalism }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = feudalismNo
					}
					always = yes
				}
			}
			OR = { PREV = { has_institution = renaissance }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = renaissanceNo
					}
					always = yes
				}
			}
			OR = { PREV = { has_institution = new_world_i }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = newWorldINo
					}
					always = yes
				}
			}
			OR = { PREV = { has_institution = printing_press }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = printingPressNo
					}
					always = yes
				}
			}
			OR = { PREV = { has_institution = global_trade }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = globalTradeNo
					}
					always = yes
				}
			}
			OR = { PREV = { has_institution = manufactories }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = manufactoriesNo
					}
					always = yes
				}
			}
			OR = { PREV = { has_institution = enlightenment }
				variable_arithmetic_trigger = {
					change_variable = {
						which = powerCost
						which = enlightenmentNo
					}
					always = yes
				}
			}
			multiply_variable = {
				which = powerCost
				value = $value$
			}
			check_variable = {
				which = power
				which = powerCost
			}
		}
}
