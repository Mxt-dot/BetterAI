	id = building_events.1
	title = building_events.1.t

		NOT = {

				trade_goods = ivory
				trade_goods = chinaware

				trade_goods = coffee
				trade_goods = cotton
				trade_goods = sugar
				trade_goods = tobacco
				trade_goods = dyes
				trade_goods = silk
				trade_goods = glass
				trade_goods = paper
				trade_goods = gems

				NOT = {
					has_construction = building
				}
			}
			add_building_construction = {
				building = manufactory
				speed = 1
				cost = 1
			}
		}
	}

country_event = {
	# building trigger manpower
	id = building_events.2
	title = building_events.2.t
	desc = building_events.2.d
	picture = INVENTION_eventPicture
	hidden = yes

	trigger = {
		ai = yes
		NOT = {
			num_of_loans = 1
		}
		treasury = 1000
		current_income_balance = 20
		NOT  ={
			manpower_percentage = 0.3
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				OR = {
					can_build = barracks
					can_build = training_fields
				}
				
				NOT = {
					has_construction = building
				}
			}
			if = {
				limit = {
					can_build = barracks
					NOT = {
						can_build = training_fields
					}
				}
				add_building_construction = {
					building = barracks
					speed = 1
					cost = 1
				}
			}
			if = {
				limit = {
					can_build = training_fields
				}
				add_building_construction = {
					building = training_fields
					speed = 1
					cost = 1
				}
			}
		}
	}

	option = {
		
	}
}
