	immediate = {
		random_owned_province = {
			limit = {
				# OR = {

				# 	trade_goods = cloth
				# 	trade_goods = salt
				# 	trade_goods = copper
				# 	trade_goods = iron
				# 	trade_goods = ivory

				# }

					AND = {
						has_building = temple
						NOT = {
							num_free_building_slots = 1
						}

							NOT = {
								has_building = farm_estate
							}
							NOT = {
								has_building = wharf
							}
							NOT = {
								has_building = weapons
							}
							NOT = {
								has_building = textile
							}
							NOT = {
								has_building = plantations
							}
							NOT = {
								has_building = tradecompany
							}
							NOT = {
								has_building = mills
							}
							NOT = {
								has_building = furnace
							}
						}
					}

						can_build = farm_estate
						can_build = wharf
						can_build = weapons
						can_build = textile
						can_build = plantations
						can_build = tradecompany
						can_build = mills
						can_build = furnace
					}
				}

					has_construction = building
				}
			}

				limit = {
					has_building = temple
					NOT = {
						num_free_building_slots = 1
					}
				}

			}
			if = {
				limit = {
					OR = {
						can_build = farm_estate
						can_build = wharf
						can_build = weapons
						can_build = textile
						can_build = plantations
						can_build = tradecompany
						can_build = tradecompany
						can_build = mills
						can_build = furnace
					}
				}
			}
			else = {
				add_building = temple
			}

			if = {
				limit = {
					can_build = farm_estate
				}
				add_building_construction = {
					building = farm_estate
					speed = 1
					cost = 1
				}
			}
			else_if = {
				limit = {
					can_build = wharf
				}
				add_building_construction = {
					building = wharf
					speed = 1
					cost = 1
				}
			}
			else_if = {
				limit = {
					can_build = weapons
				}
				add_building_construction = {
					building = weapons
					speed = 1
					cost = 1
				}
			}
			else_if = {
				limit = {
					can_build = textile
				}
				add_building_construction = {
					building = textile
					speed = 1
					cost = 1
				}
			}
			else_if = {
				limit = {
					can_build = plantations
				}
				add_building_construction = {
					building = plantations
					speed = 1
					cost = 1
				}
			}
			else_if = {
				limit = {
